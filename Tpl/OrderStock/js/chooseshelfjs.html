function ajaxGetChildrenList(id,next_model_name,model_name,choose_id,num){ var data={}; if(model_name){ data['model_name']=model_name; } if(id){ data['id']=id; } if(!next_model_name){ next_model_name="warehouse"; } if(!num){ num=''; } $.post("{:U('OrderStock/ajaxGetChildrenList')}",data,function(msg){ if(msg.status && msg.data){ $("#"+next_model_name+"_select"+num).html(''); if(msg.data.length>0){ for (var i = 0; i < msg.data.length; i++) { if(choose_id == msg.data[i].id){ $("#"+next_model_name+"_select"+num).append(''); }else{ $("#"+next_model_name+"_select"+num).append(''); } }; } }else{ $("#"+next_model_name+"_select"+num).html(""); $("#"+next_model_name+"_select"+num).append(""); } }).error(function(){ alertWarning("网络错误,请重试"); }) }
